        function App() {
            const [geracaoAtual, setGeracaoAtual] = useState("");
            const [consumoBase, setConsumoBase] = useState("");
            const [view, setView] = useState('categories'); 
            const [activeCategory, setActiveCategory] = useState(null);
            const [counts, setCounts] = useState({});
            const [toggles, setToggles] = useState([]);
            const [searchTerm, setSearchTerm] = useState("");

            // Carregamento inicial do progresso do usuário via LocalStorage usando chave única do Bio Tec
            useEffect(() => {
                const saved = localStorage.getItem('biotec_solar_v1');
                if (saved) {
                    const parsed = JSON.parse(saved);
                    setGeracaoAtual(parsed.geracaoAtual || "");
                    setConsumoBase(parsed.consumoBase || "");
                    setCounts(parsed.counts || {});
                    setToggles(parsed.toggles || []);
                }
            }, []);

            // Salvar automaticamente a cada alteração
            useEffect(() => {
                localStorage.setItem('biotec_solar_v1', JSON.stringify({
                    geracaoAtual, consumoBase, counts, toggles
                }));
            }, [geracaoAtual, consumoBase, counts, toggles]);

            const clearAll = () => {
                if(confirm("Deseja realmente limpar toda a simulação?")) {
                    setGeracaoAtual("");
                    setConsumoBase("");
                    setCounts({});
                    setToggles([]);
                    setSearchTerm("");
                    setView('categories');
                }
            };

            const consumoUpgrade = useMemo(() => {
                let total = 0;
                Object.keys(counts).forEach(id => {
                    const item = APPLIANCES.find(a => a.id === id);
                    if (item) total += (counts[id] * item.kwh);
                });
                toggles.forEach(id => {
                    const item = APPLIANCES.find(a => a.id === id);
                    if (item) total += item.kwh;
                });
                return total;
            }, [counts, toggles]);

            const numGeracao = Number(geracaoAtual) || 0;
            const numConsumo = Number(consumoBase) || 0;
            const consumoTotal = numConsumo + consumoUpgrade;
            const saldo = numGeracao - consumoTotal;
            const isOverloaded = saldo < 0;
            
            // Módulos estimados baseando-se em geração real média de 45kWh p/ placa de 450Wp
            const modulosNecessarios = isOverloaded ? Math.ceil(Math.abs(saldo) / 45) : 0;
            const percentUso = numGeracao > 0 ? Math.min(Math.round((consumoTotal / numGeracao) * 100), 250) : 0;

            const updateCount = (id, delta) => {
                setCounts(prev => ({
                    ...prev,
                    [id]: Math.max(0, (prev[id] || 0) + delta)
                }));
            };

            const toggleItem = (id) => {
                setToggles(prev => prev.includes(id) ? prev.filter(i => i !== id) : [...prev, id]);
            };

            const selectCategory = (cat) => {
                setActiveCategory(cat);
                setView('items');
                setSearchTerm("");
            };

            const getCategoryCount = (cat) => {
                return APPLIANCES
                    .filter(a => a.category === cat)
                    .reduce((acc, curr) => {
                        if (curr.multi) return acc + (counts[curr.id] || 0);
                        return acc + (toggles.includes(curr.id) ? 1 : 0);
                    }, 0);
            };

            const totalNovosEquipamentos = useMemo(() => {
                let count = 0;
                APPLIANCES.forEach(item => {
                    if (item.multi) count += (counts[item.id] || 0);
                    else if (toggles.includes(item.id)) count += 1;
                });
                return count;
            }, [counts, toggles]);

            const handleWhatsAppClick = () => {
                let listText = "";
                APPLIANCES.forEach(item => {
                    if (item.multi && counts[item.id] > 0) {
                        listText += `\n• ${counts[item.id]}x ${item.label} (+${counts[item.id] * item.kwh} kWh)`;
                    } else if (!item.multi && toggles.includes(item.id)) {
                        listText += `\n• ${item.label} (+${item.kwh} kWh)`;
                    }
                });

                const msgResult = isOverloaded 
                    ? `🚨 *Gargalo Identificado:* Sobrecarga de ${Math.abs(saldo)} kWh/mês. \n🛠️ *Ajuste recomendado:* +${modulosNecessarios} módulos fotovoltaicos.`
                    : `✅ *Suporte estável:* Sistema atual atende o novo consumo com margem.`;

                const text = `${WHATSAPP_MESSAGE}\n\n` +
                    `📊 *RESUMO DO CONSUMO SOLAR (BIO TEC):*\n` +
                    `• Geração Solar Atual: ${numGeracao} kWh\n` +
                    `• Consumo Mensal Base: ${numConsumo} kWh\n` +
                    `• Carga de Expansão Adicionada: +${consumoUpgrade} kWh\n` +
                    `• Consumo Estimado Total: ${consumoTotal} kWh\n` +
                    `\n📦 *NOVOS EQUIPAMENTOS:*${listText || "\n- Nenhum selecionado"}\n\n` +
                    `${msgResult}\n\n` +
                    `Solicito análise técnica detalhada para prosseguirmos com este projeto de upgrade de energia!`;

                window.open(`https://wa.me/${WHATSAPP_NUMBER}?text=${encodeURIComponent(text)}`, '_blank');
            };

            const pieData = useMemo(() => {

                    data.push({ name: 'Expansão', value: consumoUpgrade, color: '#f97316' });
                }

                    data.push({ name: 'Crédito Livre', value: freeSpace, color: '#10b981' });

                    return APPLIANCES.filter(a => a.category === activeCategory);

                return APPLIANCES.filter(a =>

                    a.category.toLowerCase().includes(searchTerm.toLowerCase())

                                <Icon name="sun" className="w-7 h-7" />

                            <div>
                                <h1 className="text-xl font-extrabold text-slate-100 tracking-tight">Bio Tec</h1>
                                <p className="text-xs font-semibold text-orange-500 uppercase tracking-widest">Upgrade de Energia</p>
                            </div>
                        </div>
                        <button 
                            onClick={clearAll} 
                            title="Resetar simulador"
                            className="p-3 bg-slate-950 text-slate-400 hover:text-red-400 hover:bg-red-950/40 rounded-2xl transition-all active:scale-95"
                        > 
                            <Icon name="trash" className="w-5 h-5" />
                        </button>
                    </header>

                    {/* Inputs de Entrada (Geração e Consumo) */}
                    <section className="bg-slate-900 rounded-3xl p-6 border border-slate-800 shadow-xl space-y-4 animate-slide-up">
                        <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                            <div className="bg-slate-950 p-4 rounded-2xl border border-slate-800/80 focus-within:ring-2 focus-within:ring-orange-500/20 focus-within:border-orange-500 transition-all">
                                <div className="flex items-center gap-2 text-amber-500 mb-1">
                                    <Icon name="sun" className="w-4 h-4" />
                                    <label className="text-[10px] font-black uppercase tracking-wide">Geração Solar Atual</label>
                                </div>
                                <div className="flex items-baseline gap-1">
                                    <input
                                        type="number" 
                                        placeholder="Ex: 500" 
                                        value={geracaoAtual}
                                        onChange={(e) => setGeracaoAtual(e.target.value)}
                                        className="w-full bg-transparent font-black text-2xl text-slate-100 outline-none"

                                </div>

                            {/* Barra de Pesquisa Rápida */}

                                <span className="absolute inset-y-0 left-0 pl-3.5 flex items-center pointer-events-none text-slate-500">
                                    <Icon name="search" className="w-4 h-4" />
                                </span>
                                <input
                                    type="text"
                                    placeholder="Buscar aparelho, ex: micro-ondas..."
                                    value={searchTerm}
                                    onChange={(e) => {
                                        setSearchTerm(e.target.value);
                                        if (e.target.value && view === 'categories') {
                                            setView('items');
                                        } else if (!e.target.value && !activeCategory) {
                                            setView('categories');
                                        }

                                                <div className="p-3 bg-slate-900 text-slate-400 group-hover:text-orange-500 rounded-xl border border-slate-800 shadow-sm transition-colors">
                                                    <Icon name={cat === 'Cozinha' ? 'microwave' : cat === 'Banho' ? 'droplets' : cat === 'Climatização' ? 'snowflake' : cat === 'Mobilidade' ? 'car' : cat === 'Lazer' ? 'waves' : cat === 'Beleza' ? 'sparkles' : cat === 'Ferramentas' ? 'hammer' : 'refresh-cw'} className="w-6 h-6" />
                                                </div>
                                                <span className="font-bold text-xs text-slate-300 group-hover:text-slate-100">{cat}</span>
                                            </button>
                                        );
                                    })}
                                </div>
                            </div>
                        ) : (
                            <div className="space-y-4">
                                <div className="flex justify-between items-center">
                                    <button 
                                        onClick={() => {
                                            setView('categories');
                                            setActiveCategory(null);
                                            setSearchTerm("");
                                        }}
                                        className="flex items-center gap-1.5 text-orange-500 hover:text-orange-400 font-extrabold text-xs uppercase tracking-wider transition-colors"
                                    > 
                                        <Icon name="chevron-left" className="w-4 h-4" /> Voltar
                                    </button>
                                    <span className="text-xs font-bold text-slate-400 bg-slate-950 px-3 py-1 rounded-full">
                                        {searchTerm ? "Resultados de busca" : activeCategory}
                                    </span>
                                </div>

                                <div className="space-y-2.5">
                                    {filteredAppliances.length > 0 ? (
                                        filteredAppliances.map(item => {
                                            const isSelected = item.multi ? (counts[item.id] > 0) : toggles.includes(item.id);
                                            return (
                                                <div 
                                                    key={item.id}
                                                    className={`flex items-center justify-between p-4 rounded-2xl border transition-all ${
                                                        isSelected 
                                                            ? 'bg-orange-950/20 border-orange-500/40 ring-1 ring-orange-500/20' 
                                                            : 'bg-slate-950/60 border-slate-800'
                                                    }`}
                                                >
                                                    <div className="flex items-center gap-3">
                                                        <div className={`p-2.5 rounded-xl ${isSelected ? 'bg-orange-500 text-white' : 'bg-slate-900 text-slate-400 border border-slate-800'}`}>
                                                            <Icon name={item.icon} className="w-5 h-5" />
                                                        </div>
                                                        <div>
                                                            <div className="flex items-center gap-2">
                                                                <h4 className="text-sm font-bold text-slate-200">{item.label}</h4>
                                                                <span className="text-[10px] bg-slate-800 text-slate-400 px-1.5 py-0.5 rounded-md font-semibold">
                                                                    +{item.kwh} kWh/mês
                                                                </span>
                                                            </div>
                                                            <p className="text-[10px] text-slate-500 font-medium">{item.desc}</p>
                                                        </div>
                                                    </div>

                                                    {item.multi ? (
                                                        <div className="flex items-center gap-2 bg-slate-900 border border-slate-800 rounded-xl p-1 shadow-sm">
                                                            <button 
                                                                onClick={() => updateCount(item.id, -1)} 
                                                                className="w-7 h-7 flex items-center justify-center rounded-lg bg-slate-950 hover:bg-red-950 hover:text-red-400 font-bold text-slate-400 transition-colors"
                                                            >
                                                                -
                                                            </button>
                                                            <span className="w-6 text-center font-black text-sm text-slate-200">
                                                                {counts[item.id] || 0}
                                                            </span>
                                                            <button 
                                                                onClick={() => updateCount(item.id, 1)} 
                                                                className="w-7 h-7 flex items-center justify-center rounded-lg bg-slate-950 hover:bg-orange-950 hover:text-orange-500 font-bold text-slate-400 transition-colors"
                                                            >
                                                                +
                                                            </button>
                                                        </div>
                                                    ) : (
                                                        <button
                                                            onClick={() => toggleItem(item.id)}
                                                            className={`px-3 py-1.5 rounded-xl text-xs font-bold transition-all ${
                                                                isSelected 
                                                                    ? 'bg-orange-500 text-white shadow-sm' 
                                                                    : 'bg-slate-900 border border-slate-800 text-slate-300 hover:bg-slate-800'
                                                            }`}
                                                        >
                                                            {isSelected ? 'Adicionado' : 'Adicionar'}
                                                        </button>
                                                    )}
                                                </div>
                                            );
                                        })
                                    ) : (
                                        <div className="text-center py-8 text-slate-500">
                                            <p className="text-sm font-semibold">Nenhum aparelho encontrado.</p>
                                            <p className="text-xs">Tente buscar por termos genéricos como "clima" ou "geladeira".</p>
                                        </div>
                                    )}
                                </div>
                            </div>
                        )}
                    </section>

                    {/* Dashboard Visual de Análise & Gráfico */}
                    <section className="bg-slate-900 rounded-3xl p-6 border border-slate-800 shadow-xl space-y-6 animate-slide-up">
                        <div className="flex justify-between items-start">
                            <div>
                                <h3 className="text-xs font-black text-slate-500 uppercase tracking-widest mb-1">Carga Estimada</h3>
                                <div className="flex items-baseline gap-1">
                                    <span className={`text-4xl font-black ${isOverloaded ? 'text-red-400' : 'text-slate-100'}`}>
                                        {percentUso}%
                                    </span>
                                    <span className="text-xs text-slate-500 font-bold">do suporte solar</span>
                                </div>
                            </div>
                            <div className="text-right">
                                <h3 className="text-xs font-black text-slate-500 uppercase tracking-widest mb-1">Nova Carga</h3>
                                <div className="flex items-baseline gap-1 justify-end">
                                    <span className="text-2xl font-black text-orange-500">+{consumoUpgrade}</span>
                                    <span className="text-xs text-slate-500 font-bold">kWh/mês</span>
                                </div>
                            </div>
                        </div>

                        {/* Barra de Progresso Simples */}
                        <div className="h-3 w-full bg-slate-950 rounded-full overflow-hidden">
                            <div 
                                className={`h-full transition-all duration-500 ease-out ${isOverloaded ? 'bg-red-500' : 'bg-emerald-500'}`}
                                style={{ width: `${Math.min(percentUso, 100)}%` }}
                            />
                        </div>

                        {/* Donut Chart Visual do Balanço de Energia */}
                        {(numConsumo > 0 || consumoUpgrade > 0) && (
                            <div className="h-44 w-full">
                                <ResponsiveContainer width="100%" height="100%">
                                    <PieChart>
                                        <Pie
                                            data={pieData}
                                            innerRadius={50}
                                            outerRadius={70}
                                            paddingAngle={3}
                                            dataKey="value"
                                        >
                                            {pieData.map((entry, index) => (
                                                <Cell key={`cell-${index}`} fill={entry.color} />
                                            ))}
                                        </Pie>
                                        <Tooltip 
                                            contentStyle={{ backgroundColor: '#090d16', borderColor: '#1e293b' }} 
                                            itemStyle={{ color: '#f1f5f9' }}
                                            formatter={(value) => [`${value} kWh`, null]} 
                                        />
                                    </PieChart>
                                </ResponsiveContainer>
                                <div className="flex justify-center flex-wrap gap-x-4 gap-y-1 text-[11px] font-bold text-slate-400">
                                    {pieData.map((d, i) => (
                                        <span key={i} className="flex items-center gap-1.5">
                                            <span className="w-2.5 h-2.5 rounded-full" style={{ backgroundColor: d.color }} />
                                            {d.name}: {d.value} kWh
                                        </span>
                                    ))}
                                </div>
                            </div>
                        )}

                        {/* Recomendações & CTA "Quero mais energia" */}
                        {isOverloaded ? (
                            <div className="bg-red-950/20 border border-red-900/40 rounded-2xl p-5 space-y-4">
                                <div className="flex gap-3">
                                    <div className="p-2 bg-red-900/30 text-red-400 rounded-xl h-fit">
                                        <Icon name="zap" className="w-5 h-5" />
                                    </div>
                                    <div>
                                        <h4 className="font-extrabold text-sm text-red-200">Seu sistema atual entrará em déficit</h4>
                                        <p className="text-xs text-red-400 leading-relaxed mt-0.5">
                                            A adição de novos aparelhos superará a geração mensal em <strong className="font-extrabold">{Math.abs(saldo)} kWh/mês</strong>. Recomenda-se expandir o arranjo fotovoltaico para evitar aumento na fatura.
                                        </p>
                                    </div>
                                </div>

                                <div className="bg-slate-950/80 border border-red-900/30 rounded-xl p-4 flex justify-between items-center shadow-sm">
                                    <div>
                                        <span className="text-[10px] font-black text-slate-500 uppercase tracking-wide">Módulos Recomendados</span>
                                        <p className="text-lg font-black text-slate-200">+{modulosNecessarios} placas de 450Wp</p>
                                    </div>
                                    <div className="bg-red-600 text-white font-black text-sm px-3.5 py-1.5 rounded-xl shadow-md">
                                        Upgrade
                                    </div>
                                </div>

                                <button
                                    onClick={handleWhatsAppClick}
                                    className="w-full bg-gradient-to-r from-orange-500 to-amber-600 hover:from-orange-600 hover:to-amber-700 text-white font-extrabold py-4 px-6 rounded-2xl flex items-center justify-center gap-2 shadow-lg shadow-orange-950/50 transition-all active:scale-[0.98]"
                                >
                                    <span>Quero mais energia</span>
                                    <Icon name="arrow-right" className="w-4 h-4" />
                                </button>
                            </div>
                        ) : (
                            <div className="p-5 bg-emerald-950/20 border border-emerald-900/40 rounded-2xl space-y-4">
                                <div className="flex gap-3">
                                    <div className="p-2 bg-emerald-900/30 text-emerald-400 rounded-xl h-fit">
                                        <Icon name="check-circle" className="w-5 h-5" />
                                    </div>
                                    <div>
                                        <h4 className="font-extrabold text-sm text-emerald-200">Excelente! Sistema dimensionado</h4>
                                        <p className="text-xs text-emerald-400 leading-relaxed mt-0.5">
                                            Seu arranjo de inversores e painéis atuais possui folga de <strong className="font-extrabold">{numGeracao > 0 ? Math.max(0, saldo) : 0} kWh/mês</strong> e suportará com tranquilidade as novas adições inseridas.
                                        </p>
                                    </div>
                                </div>
                                
                                {totalNovosEquipamentos > 0 ? (
                                    <button
                                        onClick={handleWhatsAppClick}
                                        className="w-full bg-gradient-to-r from-emerald-500 to-teal-600 hover:from-emerald-600 hover:to-teal-700 text-white font-extrabold py-3.5 px-6 rounded-2xl flex items-center justify-center gap-2 shadow-lg shadow-emerald-950/50 transition-all active:scale-[0.98]"
                                    >
                                        <span>Quero mais energia</span>
                                        <Icon name="arrow-right" className="w-4 h-4" />
                                    </button>
                                ) : (
                                    <button
                                        onClick={handleWhatsAppClick}
                                        className="w-full bg-slate-850 hover:bg-slate-800 border border-slate-700 text-slate-200 font-extrabold py-3.5 px-6 rounded-2xl flex items-center justify-center gap-2 transition-all active:scale-[0.98]"
                                    >
                                        <span>Quero mais energia</span>
                                        <Icon name="arrow-right" className="w-4 h-4" />
                                    </button>
                                )}
                            </div>
                        )}
                    </section>
                </div>
            );
        }

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
